Rate of Change per year (using linear regression slope)

compute_slope <- function(air_quality_deu, pollutant) {
  model <- lm(air_quality_deu[[pollutant]] ~ air_quality_deu$year)
  slope <- coef(model)[2]  # yearly rate of change
  return(slope)
}
